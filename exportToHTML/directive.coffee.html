37                       when 'single' 
38                           scope.singleItemActive = dataCache.get('singleItemActive' + guid) or 0

78

83                   dataCache.put('doubleItem' + guid, { 
84                       doubleLeftItem: contentObj

88                       leftItem: contentObj 
89                       rightItem: condition 
90                   }

95                   contentIndex = scope.contentList.indexOf(contentObj) 
96                   conditionIndex = scope.conditionTags.indexOf(condition) 
97                   if contentIndex is -1 
98                       scope.contentList.push(contentObj) 
99                       scope.conditionTags.push(condition) 
100                  else 
101                      if condition.code isnt '1' 
102                          if conditionIndex isnt -1 
103                              scope.conditionTags[conditionIndex] = condition 
104                          else 
105                              scope.conditionTags.push(condition) 
106                      else 
107                          if conditionIndex isnt -1 
108                              scope.conditionTags.splice(conditionIndex, 1) 
109                          else 
110                              return 
111                  return 
112   
113              scope.emptyHandler = (e)-> 
114                  e.stopPropagation() 
115                  scope.conditionTags.length = 0 
116                  return 
117   
118              scope.stopPropagation = (e)-> 
119                  e.stopPropagation() 
120                  return 
121              scope.cancelHandler = -> 
122                  hideContent() 
123                  return

126                  return 
127   
128              scope.confirmHandler = -> 
129                  emitData(exportData)

131                  return 
132   
133              emitData = (data)->

135                  return 
136              return 
137      }
